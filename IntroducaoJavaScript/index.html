/**
 * EXEMPLO 2: Funções em JavaScript
 */

// 1. Função Tradicional (Named Function)
// Possui hoisting (pode ser chamada antes de declarada)
function saudar(nome) {
    return `Olá, ${nome}! Seja bem-vindo ao aprendizado de JavaScript.`;
}

// 2. Expressão de Função (Function Expression)
const calcularIdade = function(anoNascimento) {
    const anoAtual = new Date().getFullYear();
    return anoAtual - anoNascimento;
};

// 3. Arrow Function (Funções de Seta - ES6+)
// Retorno implícito de uma única linha
const somar = (a, b) => a + b;

const converterParaMaiusculas = texto => texto.toUpperCase();

// Arrow function composta
const gerarSaudacaoPersonalizada = (nome, anoNascimento) => {
    const idade = calcularIdade(anoNascimento);
    const saudacao = saudar(nome);
    return `${saudacao} Você tem ${idade} anos.`;
};
/**
 * EXEMPLO 3: Estruturas de Decisão e Repetição
 */

// 1. Estruturas Condicionais (if / else if / else)
function classificarIdade(idade) {
    if (idade < 0) {
        return "Idade inválida";
    } else if (idade < 12) {
        return "Criança";
    } else if (idade < 18) {
        return "Adolescente";
    } else if (idade < 60) {
        return "Adulto";
    } else {
        return "Idoso";
    }
}

// 2. Estrutura Switch
function traduzirCor(corIngles) {
    switch (corIngles.toLowerCase()) {
        case "red": return "Vermelho";
        case "blue": return "Azul";
        case "green": return "Verde";
        case "yellow": return "Amarelo";
        default: return "Cor desconhecida";
    }
}

// 3. Estruturas de Repetição (Loops)
// Loop FOR
function gerarSequenciaNumerica(limite) {
    const numeros = [];
    for (let i = 1; i <= limite; i++) {
        numeros.push(i);
    }
    return numeros;
}

// Loop WHILE
function contagemRegressiva(inicio) {
    const passos = [];
    let contador = inicio;
    while (contador > 0) {
        passos.push(contador);
        contador--;
    }
    passos.push("Decolar! 🚀");
    return passos;
}
